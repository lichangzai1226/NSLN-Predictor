import streamlit as st import pandas as pd import numpy as np from sklearn.model_selection import train_test_split from sklearn.preprocessing import StandardScaler from sklearn.impute import KNNImputer from sklearn.linear_model import LassoCV, LogisticRegression from sklearn.metrics import roc_curve, confusion_matrix import os # ========================================== # 1. 页面配置 # ========================================== st.set_page_config( page_title="NSLN 风险分层决策系统", page_icon="⚕️", layout="centered", initial_sidebar_state="collapsed" ) st.markdown(""" """, unsafe_allow_html=True) # ========================================== # 2. 核心算法引擎 # ========================================== @st.cache_resource def train_model(): files = ['breast cancer.xlsx', 'breast cancer.xlsx - Sheet1.csv'] df = None for f in files: if os.path.exists(f): try: if f.endswith('.xlsx'): df = pd.read_excel(f) else: df = pd.read_csv(f) break except: continue if df is None: return None, None, None, None, None, None, None, None, None # --- 数据清洗 --- df.columns = df.columns.str.strip() rename_map = { 'NSLN status(1=positive,0=negative)': 'NSLN_Status', 'NSLN status(1=positive，0=negative)': 'NSLN_Status', 'Age': 'Age', 'Tumor location': 'Tumor_Location', 'Tumor size': 'Tumor_Size', 'Grade': 'Grade', 'TILS ': 'TILS', 'TILS': 'TILS', 'LVI（1=Yes，0=No）': 'LVI', 'Number of positive SLN': 'SLN_Count', 'ER(1=positive，0=negative)': 'ER', 'PR': 'PR', 'Her-2(1=positive，0=negative)': 'Her2', 'KI-67': 'Ki67', 'P53': 'P53', 'AR': 'AR', 'Molecular_Subtype': 'Molecular_Subtype', '编号 ': 'ID', '编号': 'ID' } df = df.rename(columns=rename_map) def clean_pct(x): if isinstance(x, str) and '%' in x: try: return float(x.replace('%', ''))/100 except: return np.nan return pd.to_numeric(x, errors='coerce') for c in ['TILS', 'Ki67', 'P53', 'AR']: if c in df.columns: df[c] = df[c].apply(clean_pct) num_cols = ['Age', 'Tumor_Size', 'Grade', 'LVI', 'SLN_Count', 'NSLN_Status'] for c in num_cols: if c in df.columns: df[c] = pd.to_numeric(df[c], errors='coerce') if 'NSLN_Status' not in df.columns: return None, None, None, None, None, None, None, None, None df = df.dropna(subset=['NSLN_Status']) # AR Cutoff best_cutoff = 0.3 if 'AR' in df.columns: df_ar = df.dropna(subset=['AR']) if len(df_ar) > 10: fpr, tpr, thresholds = roc_curve(df_ar['NSLN_Status'], df_ar['AR']) best_cutoff = thresholds[np.argmax(tpr - fpr)] df['AR_Cat'] = (df['AR'] >= best_cutoff).astype(int) else: df['AR_Cat'] = (df['AR'] >= 0.3).astype(int) else: df['AR_Cat'] = 0 exclude = ['编号', '住院号', '姓名', 'NSLN_Status', 'SLN（0=Micro metastasis,1=Macro metastasis）', 'AR', 'ID'] valid_cols = [] for c in df.columns: if c not in exclude: if df[c].dtype in [np.float64, np.int64] or df[c].nunique() < 10: valid_cols.append(c) X = df[valid_cols] y = df['NSLN_Status'].astype(int) cat_cols = [c for c in ['Tumor_Location', 'Molecular_Subtype'] if c in X.columns] X_enc = pd.get_dummies(X, columns=cat_cols, drop_first=True) final_features = X_enc.columns.tolist() # 划分数据集 X_train, X_val, y_train, y_val = train_test_split(X_enc, y, test_size=0.3, random_state=42, stratify=y) imputer = KNNImputer(n_neighbors=5) scaler = StandardScaler() X_train_imp = imputer.fit_transform(X_train) X_val_imp = imputer.transform(X_val) X_train_sc = scaler.fit_transform(X_train_imp) X_val_sc = scaler.transform(X_val_imp) try: lasso = LassoCV(cv=5, random_state=42).fit(X_train_sc, y_train) coefs = pd.Series(lasso.coef_, index=final_features) selected_feats = coefs[coefs != 0].index.tolist() if not selected_feats: selected_feats = final_features except: selected_feats = final_features X_train_final = pd.DataFrame(X_train_sc, columns=final_features)[selected_feats] X_val_final = pd.DataFrame(X_val_sc, columns=final_features)[selected_feats] lr_model = LogisticRegression(max_iter=1000).fit(X_train_final, y_train) y_prob_val = lr_model.predict_proba(X_val_final)[:, 1] return lr_model, selected_feats, imputer, scaler, best_cutoff, final_features, y_val, y_prob_val, final_features try: model, sel_feats, imputer, scaler, ar_cutoff, all_encoded_cols, y_val_true, y_val_prob, all_feats = train_model() except: st.error("数据加载失败"); st.stop() if model is None: st.warning("⚠️ 请上传数据文件"); st.stop() # ========================================== # 3. 界面布局 # ========================================== st.markdown('
